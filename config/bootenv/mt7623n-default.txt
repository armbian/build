board_name=mt7623-bpi-r2

stdin=serial
stdout=serial
bootdelay=1
ethact=eth0

ipaddr=10.30.0.1

mmc_device=1

f_set_bootargs=setenv bootargs console=ttyS2,115200n1 root=/dev/mmcblk${mmc_device}p${boot_part} rw rootfstype=ext4 rootwait coherent_pool=4M audit=0 ${usrargs}
f_read_kernel=ext4load mmc ${mmc_device}:${boot_part} ${loadaddr} ${kernel_file}
f_setup_pri=echo Booting primary firmware image...; setenv boot_part 1
f_setup_sec=echo Booting secondary firmare image...; setenv boot_part 2
bootcmd=mmc device ${mmc_device}; mmc init ${mmc_device}; run f_set_bootargs; run f_read_kernel; bootm
boot_pri=mmc device ${mmc_device}; run f_setup_pri; run f_set_bootargs; run f_read_kernel; bootm ${loadaddr}
boot_sec=mmc device ${mmc_device}; run f_setup_sec; run f_set_bootargs; run f_read_kernel; bootm ${loadaddr}
